	title = "Belgian Revolution"
	desc = "EVTDESC7000"
	
	trigger = {
		tag = NET

		NOT = { exists = BEL }
		any_owned_province = {
			OR = {
				is_core = WAL

			}
		}
	}

		months = 6
	}

		name = "Damn them!"

			limit = {
				OR = {
					is_core = FLA

			}
			add_core = BEL
		}
		release = BEL
		LUX = { 
			all_core = { 
				add_core = NET
				add_core = BEL
			}
			release = BEL
		}
		war = {
			target = BEL

				casus_belli = annex_core_country
			}
			call_ally = no
		}
		BEL = {
			small_arms = 100
			canned_food = 200
			artillery = 50
			luxury_clothes = 50
			wine = 50

		}
		ENG = {
			diplomatic_influence = {
				who = BEL
				value = 100
			}
		}
		set_country_flag = belgian_independence
		
		ai_chance = {
			factor = 90
		}
	}
	option = {
		name = "Let them go."
		any_owned = {
			limit = {
				OR = {
					is_core = FLA
					is_core = WLL
					province_id = 381
				}
			}
			add_core = BEL

		release = BEL

			all_core = { 
				add_core = BEL
